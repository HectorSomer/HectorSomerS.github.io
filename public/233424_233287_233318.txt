void bubble(int limite, int lista[]) {
    	bool b;	
    	int temporal = 0;
    	int cambios = 0;
    	int iteraciones = 0;
	do { 
		b = false;
		cambios=cambios+1;
		for(int i=0; i<limite-1; i++) { 
			if(lista[i]>lista[i+1]) {
				b = true;
				iteraciones=iteraciones+1;
				int temporal = lista[i];
				lista[i] = lista[i+1];
				lista[i+1] = temporal;
			} 
		}
		
		for(int c=0; c<limite; c++) {
			cout << lista[c] << "\t";
		}
		cout << endl;
	} while(b == true);
	cout << "El numero de veces que se hizo el primero ciclo fue "<< cambios;
	cout <<endl;
	cout <<"El numero de veces que se hizo el segundo ciclo fue "<< iteraciones;
	cout << endl;
}

void insertion(int limite, int lista[]){	
int repeticion=0;
int cambios=0;
	for(int i=1; i<limite; i++) {
		int it = i;
		repeticion=repeticion+1;
		for(int j=i-1; j>=0; j--) {
			if(lista[it] < lista[j]) {
				int nt = lista[it];
				lista[it] = lista[j];
				lista[j] = nt;
				cambios=cambios+1;
				it = j;
			} else {
				break;
			}
		}
		
		for(int k=0; k<limite; k++) {
			cout << lista[k] << "\t";
		}
		cout << "\n";
	}
	cout << "El número de ciclos fue "<< repeticion<<endl; 
	cout <<"El número de iteraciones fue " << cambios<< endl;
}
